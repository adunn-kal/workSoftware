477template <class dataType>
478Queue<dataType>::Queue (BaseType_t queue_size, const char* p_name,
479 TickType_t wait_time)
480 : BaseShare (p_name)
481{
482 // Create a FreeRTOS queue object with space for the data items
483 handle = xQueueCreate (queue_size, sizeof (dataType));
484
485 // Store the wait time; it will be used when writing to the queue
486 ticks_to_wait = wait_time;
487
488 // Save the buffer size
489 buf_size = queue_size;
490
491 // We haven't stored any items in the queue yet
492 max_full = 0;
493}

505template <class dataType>
506inline bool Queue<dataType>::put (const dataType item)
507{
508 bool return_value = (bool)(xQueueSendToBack (handle, &item,
509 ticks_to_wait));
510
511 // Keep track of the maximum fillage of the queue
512 uint16_t fillage = uxQueueMessagesWaiting (handle);
513 if (fillage > max_full)
514 {
515 max_full = fillage;
516 }
517
518 return (return_value);
519}

529template <class dataType>
530inline bool Queue<dataType>::ISR_put (const dataType item)
531{
532 // This value is set true if a context switch should occur due to this data
533 signed portBASE_TYPE shouldSwitch = pdFALSE;
534
535 bool return_value; // Value returned from this method
536
537 // Call the FreeRTOS function and save its return value
538 return_value = (bool)(xQueueSendToBackFromISR (handle, &item,
539 &shouldSwitch));
540
541 // Keep track of the maximum fillage of the queue. BUG: max_full isn't
542 // thread safe (but getting max_full corrupted shouldn't cause a calamity)
543 uint16_t fillage = uxQueueMessagesWaitingFromISR (handle);
544 if (fillage > max_full)
545 {
546 max_full = fillage;
547 }
548
549 // Return the return value saved from the call to xQueueSendToBackFromISR()
550 return (return_value);
551}
